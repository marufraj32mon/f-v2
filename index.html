const phone = '01603746079';
const res = await fetch(
  `YOUR_VERCEL_URL/api/check?phone=${phone}`
);
const data = await res.json();

if (data.success) {
  // data.data তে কুরিয়ার তথ্য আছে
  Object.entries(data.data).forEach(([name, info]) => {
    console.log(name, info);
  });
}

                

                

                  
                  
$phone = '01603746079';
 $url = 'YOUR_VERCEL_URL/api/check?phone=' . urlencode($phone);

 $ch = curl_init();
curl_setopt($ch, CURLOPT_URL, $url);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_TIMEOUT, 10);
 $response = curl_exec($ch);
curl_close($ch);

 $data = json_decode($response, true);
if ($data['success']) {
    foreach ($data['data'] as $name => $info) {
        echo $name . ': ' . $info['total_parcels'] . ' parcels';
    }
}
